import { db } from "../config/firebase-config.js"; import { collection, getDocs, query, where, doc, getDoc, deleteDoc } from "https://www.gstatic.com/firebasejs/10.8.0/firebase-firestore.js"; export const subscriberService = { // جلب المشتركين مع دعم صلاحية السوبر أدمن getAllSubscribers: async () => { const user = JSON.parse(localStorage.getItem('currentUser')); if (!user || !user.uid) throw new Error("غير مسجل"); // جلب بيانات المستخدم للتأكد من الرتبة const userDoc = await getDoc(doc(db, "users", user.uid)); const userData = userDoc.data(); let q; // إذا كنت السوبر أدمن، اجلب كل المشتركين من كل مكان if (userData && userData.role === 'super_admin') { q = collection(db, "subscribers"); } else { // إذا كنت مستخدماً عادياً، اجلب مشتركي مشروعك فقط q = query(collection(db, "subscribers"), where("projectId", "==", user.projectId)); } const querySnapshot = await getDocs(q); return querySnapshot.docs.map(doc => ({ id: doc.id, ...doc.data() })); }, // حذف مشترك deleteSubscriber: async (id) => { return await deleteDoc(doc(db, "subscribers", id)); } };
