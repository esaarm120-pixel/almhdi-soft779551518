// subscribers.html أو ملف السيرفس المخصص للمشتركين بداخل subscribers.html import { db } from "../config/firebase-config.js"; import { collection, getDocs, query, where, doc, getDoc, setDoc, deleteDoc } from "https://www.gstatic.com/firebasejs/10.8.0/firebase-firestore.js"; export const subscriberService = { getAllSubscribers: async () => { const user = JSON.parse(localStorage.getItem('currentUser')); if (!user?.uid) { throw new Error("يرجى تسجيل الدخول أولاً"); } const userRef = doc(db, "users", user.uid); let userSnap = await getDoc(userRef); let userData; if (!userSnap.exists()) { console.log("تفقد المشتركين: المستند مفقود بـ UID، جاري التوليف والمطابقة عبر البريد الإلكتروني للمستخدم..."); const emailQuery = query(collection(db, "users"), where("email", "==", user.email || "")); const querySnap = await getDocs(emailQuery); if (!querySnap.empty) { const originalDoc = querySnap.docs[0]; userData = originalDoc.data(); await setDoc(doc(db, "users", user.uid), { ...userData, updatedFromRandomId: originalDoc.id }); userSnap = await getDoc(userRef); } else { throw new Error("المستخدم غير موجود في قاعدة البيانات"); } } userData = userSnap.data(); let q; const role = (userData.role || "").toLowerCase(); if (role === "super_admin") { q = collection(db, "subscribers"); } else { const projId = userData.projectId || user.projectId; if (!projId) { throw new Error("projectId غير موجود في المستخدم"); } q = query( collection(db, "subscribers"), where("projectId", "==", projId) ); } const snapshot = await getDocs(q); return snapshot.docs.map(d => ({ id: d.id, ...d.data() })); }, deleteSubscriber: async (id) => { if (!id) throw new Error("ID غير موجود"); return await deleteDoc(doc(db, "subscribers", id)); } };
